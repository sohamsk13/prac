#include<iostream>
#include<vector>
#include<stack>
#include<queue>
#include<omp.h>

using namespace std;

class Graph{

    int vertices, edges;

    vector<vector<int>> graph;

    vector<bool> visited;

public:

    // Constructor for input
    Graph(){

        cout << "Enter number of vertices: ";
        cin >> vertices;

        cout << "Enter number of edges: ";
        cin >> edges;

        graph.resize(vertices);

        // Taking edges as input
        for(int i = 0; i < edges; i++){

            int a, b;

            cout << "Enter edge (u v): ";
            cin >> a >> b;

            graph[a].push_back(b);
            graph[b].push_back(a);
        }
    }

    void initializeVisited(){

        visited.assign(vertices, false);
    }

    void printGraph(){

        for(int i = 0; i < vertices; i++){

            cout << i << " -> ";

            for(auto j = graph[i].begin(); j != graph[i].end(); j++){

                cout << *j << " ";
            }

            cout << endl;
        }
    }

    // Parallel DFS
    void parallelDFS(int start){

        stack<int> s;

        s.push(start);

        visited[start] = true;

        while(!s.empty()){

            int current = s.top();

            #pragma omp critical
            s.pop();

            cout << current << " ";

            #pragma omp parallel for
            for(auto j = graph[current].begin(); j != graph[current].end(); j++){

                if(visited[*j] == false){

                    #pragma omp critical
                    {
                        s.push(*j);
                        visited[*j] = true;
                    }
                }
            }
        }
    }

    // Parallel BFS
    void parallelBFS(int start){

        queue<int> q;

        q.push(start);

        visited[start] = true;

        while(!q.empty()){

            int current = q.front();

            #pragma omp critical
            q.pop();

            cout << current << " ";

            #pragma omp parallel for
            for(auto j = graph[current].begin(); j != graph[current].end(); j++){

                if(visited[*j] == false){

                    #pragma omp critical
                    {
                        q.push(*j);
                        visited[*j] = true;
                    }
                }
            }
        }
    }
};

int main(){

    Graph g;

    cout << "\nAdjacency List:\n";
    g.printGraph();

    cout << "\nParallel DFS:\n";
    g.initializeVisited();
    g.parallelDFS(0);

    cout << "\n\nParallel BFS:\n";
    g.initializeVisited();
    g.parallelBFS(0);

    return 0;
}
